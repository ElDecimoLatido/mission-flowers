    // Velocidades (ms)

    const TYPE_SPEED = 55;      // velocidad de letras

    const LINE_DELAY = 450;     // pausa entre líneas

    const BOOT_DELAY = 650;     // pausa inicial


    // ==== Motor de tecleo ====

    const screen = document.getElementById("screen");

    const replayBtn = document.getElementById("replay");


    function sleep(ms){ return new Promise(r => setTimeout(r, ms)); }


    // ✅ Auto-scroll: siempre baja al final

    function scrollToBottom() {

      screen.scrollTop = screen.scrollHeight;

    }


    async function typeAll(){

      replayBtn.style.display = "none";

      screen.textContent = "";

      const cursor = document.createElement("span");

      cursor.className = "cursor";

      screen.appendChild(cursor);


      await sleep(BOOT_DELAY);


      for (const line of LINES){

        for (let i = 0; i < line.length; i++){

          cursor.insertAdjacentText("beforebegin", line[i]);

          scrollToBottom(); // ✅ baja conforme escribe

          await sleep(TYPE_SPEED + Math.random()*12);

        }

        cursor.insertAdjacentText("beforebegin", "\n");

        scrollToBottom(); // ✅ baja al cambiar de línea

        await sleep(LINE_DELAY);

      }


      replayBtn.style.display = "inline-block";

      scrollToBottom();

    }


    replayBtn.addEventListener("click", (e) => {

      e.preventDefault();

      typeAll();

    });


    // Auto-inicio al abrir

    typeAll();
